Exemplo de algoritmo, usando o pacote \texttt{algorithm2e} (Algoritmo~\ref{alg:algoritmo})
\begin{algorithm}
    \caption{Exemplo de algoritmo.}
    \Entrada{$x$ e $y$ inteiros.}
    \Saida{$z$ inteiro.}
    \Se{$x>y$}{
    	z = x - y\;
    }
    \label{alg:algoritmo}
\end{algorithm}
